toothbrush.js

Stream-based mustache template rendering for node.js

Installation

npm install toothbrush

Usage

var toothbrush = require("toothbrush");

http.createSever(function(req, res){
    toothbrush("./test.html", res, {"title": "toothbrush"});
}).listen(8080);
